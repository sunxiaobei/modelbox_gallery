178 template <typename T>
179 void BufferManageView::GenProcessInfo(
180  const std::unordered_map<std::string, T> &parent_data,
181  size_t data_count_per_port,
182  const std::function<std::shared_ptr<Buffer>(const T &container, size_t idx)>
183  &get_buffer_at,
184  std::vector<std::shared_ptr<BufferProcessInfo>> &process_info_list,
185  bool all_in_one_process) {
186  if (all_in_one_process) {
187  process_info_list.push_back(std::make_shared<BufferProcessInfo>());
188  } else {
189  process_info_list.reserve(data_count_per_port);
190  for (size_t i = 0; i < data_count_per_port; ++i) {
191  process_info_list.push_back(std::make_shared<BufferProcessInfo>());
192  }
193  }
194 
195  for (auto &port_data_item : parent_data) {
196  auto &port_name = port_data_item.first;
197  auto &data_list = port_data_item.second;
198  auto index_info_list =
199  std::make_shared<std::list<std::shared_ptr<BufferIndexInfo>>>();
200  for (size_t i = 0; i < data_count_per_port; ++i) {
201  auto buffer = get_buffer_at(data_list, i);
202  auto index_info = BufferManageView::GetIndexInfo(buffer);
203  index_info_list->push_back(index_info);
204  if (!all_in_one_process) {
205  auto inherit_info = process_info_list[i];
206  inherit_info->SetParentBuffers(port_name, std::move(*index_info_list));
207  index_info_list->clear();
208  }
209  }
210  if (all_in_one_process) {
211  auto inherit_info = process_info_list.front();
212  inherit_info->SetParentBuffers(port_name, std::move(*index_info_list));
213  }
214  }
215 }
