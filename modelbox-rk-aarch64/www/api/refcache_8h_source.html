221  std::shared_ptr<T> GetLocked(std::unique_lock<std::mutex> *lock, KEY &key,
222  bool is_block = false) {
223  bool is_success = false;
224  std::shared_ptr<bool> guard(&is_success, [&](bool *result) {
225  if (is_success == false) {
226  ReleaseLocked(key);
227  }
228  });
229 
230  auto itr = ref_data_set_.find(key);
231  if (itr == ref_data_set_.end()) {
232  return nullptr;
233  }
234 
235  auto ref = itr->second;
236  ref->refcount_ += 1;
237 
238  if (ref->data_ == nullptr) {
239  if (is_block == false) {
240  return nullptr;
241  }
242 
243  if (ref->trans_ == nullptr) {
244  return nullptr;
245  }
246 
247  cond_.wait(*lock, [&]() {
248  return ref->data_ != nullptr || ref->trans_ == nullptr;
249  });
250

255

259  }
260  }
261 
262  is_success = true;
263  std::shared_ptr<T> ret(ref->data_.get(),
264  [this, key](void *ptr) { Release(key); });
