103  template <typename T>
104  Status Build(const std::vector<std::vector<size_t>>& shape_list) {
105  std::vector<size_t> data_size_list(shape_list.size(), 0);
106  std::transform(shape_list.begin(), shape_list.end(), data_size_list.begin(),
107  [](const std::vector<size_t>& shape) {
108  return Volume(shape) * sizeof(T);
109  });
110 
111  size_t size = std::accumulate(data_size_list.begin(), data_size_list.end(),
112  (size_t)0, std::plus<size_t>());
113  if (!bl_->dev_mem_) {
114  return {STATUS_INVALID, "device memory must not be nullptr."};
115  }
116 
117  auto device = bl_->dev_mem_->GetDevice();
118  bl_->dev_mem_ = device->MemAlloc(size);
119  if (!bl_->dev_mem_) {
120  MBLOG_WARN << " MemAlloc " << size << " byte data failed";
121  return STATUS_NOMEM;
122  }
123 
124  bl_->buffer_list_.resize(data_size_list.size(), nullptr);
125 
126  size_t offset = 0;
127  for (size_t i = 0; i < bl_->buffer_list_.size(); i++) {
128  auto&& mem = bl_->dev_mem_->Cut(offset, data_size_list[i]);
129  bl_->buffer_list_[i] = std::make_shared<TensorBuffer>(mem);
130  offset += data_size_list[i];
131  }
132 
133  bl_->is_contiguous_ = true;
134  return SetShape<T>(shape_list);
135  }
